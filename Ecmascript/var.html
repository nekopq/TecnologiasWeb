Identificadores var, let, const

Existen otros identificadores para almacenar valores: let y const.

Al usar var:

var snack = 'Meow Mix';
function getFood(food) {
if (food) {
var snack = 'Friskies';
return snack;
}
return snack;
}
getFood(false); // undefined

Al usar let:

let snack = 'Meow Mix';
function getFood(food) {
if (food) {
let snack = 'Friskies';
return snack;
}
return snack;
}
getFood(false); // 'Meow Mix'

Al usar let o const:

console.log(x); // ReferenceError: x is not defined
let x = 'hi';
